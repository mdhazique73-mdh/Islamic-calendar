        const [time, modifier] = timeStr.split(' ');

        let [hours, minutes] = time.split(':');

       

        if (hours === '12') hours = '00';

        if (modifier === 'PM') hours = parseInt(hours) + 12;

       

        const date = new Date(dateRef);

        date.setHours(parseInt(hours), parseInt(minutes), 0, 0);

        return date;

    }

    function formatTimeRemaining(ms) {

        if (ms <= 0) return "00:00:00";

        const seconds = Math.floor((ms / 1000) % 60);

        const minutes = Math.floor((ms / (1000 * 60)) % 60);

        const hours = Math.floor((ms / (1000 * 60 * 60)));

        return `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

    }

    function formatDateLong(dateObj) {

        const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };

        return dateObj.toLocaleDateString('en-US', options);

    }

    function renderList() {

        const listEl = document.getElementById('dateList');

        listEl.innerHTML = '';

        ramadanData.forEach((item, index) => {

            const li = document.createElement('li');

            let classes = 'date-item';

            if (index === currentSelectedIndex) classes += ' active';

            if (index === todayIndex) classes += ' today';

           

            li.className = classes;

            li.onclick = () => selectDay(index);

            li.innerHTML = `

                <div class="item-left">

                    <span class="roza-num">ROZA ${item.day}</span>

                    <span class="date-str">${item.date}</span>

                    <span class="day-name">${item.dayName}</span>

                </div>

                <div class="item-right">

                    <div class="item-time">

                        <span class="time-label">Sehri</span>

                        <span class="time-val">${item.sehri}</span>

                    </div>

                    <div class="item-time">

                        <span class="time-label">Iftar</span>

                        <span class="time-val">${item.iftar}</span>

                    </div>

                </div>

            `;

            listEl.appendChild(li);

        });

    }

    function updateCard() {

        const data = ramadanData[currentSelectedIndex];

        const dateObj = new Date(data.date + " 00:00:00");

       

        document.getElementById('displayDate').textContent = formatDateLong(dateObj);

        document.getElementById('displayRoza').textContent = `ROZA ${data.day}`;

        document.getElementById('displaySehri').textContent = data.sehri;

        document.getElementById('displayIftar').textContent = data.iftar;

        renderList();

    }

    function selectDay(index) {

        currentSelectedIndex = index;

        updateCard();

    }

    function jumpToToday() {

        selectDay(todayIndex);

        setTimeout(() => {

            const list = document.getElementById('dateList');

            const activeItem = list.children[todayIndex];

            if(activeItem) {

                activeItem.scrollIntoView({ behavior: 'smooth', block: 'center' });

            }

        }, 100);

    }

    function updateCountdown() {

        const now = new Date();

        const isViewingToday = (currentSelectedIndex === todayIndex);

        let data, sehriTime, iftarTime, dateRef;

        if (isViewingToday) {

            data = ramadanData[todayIndex];

            dateRef = now;

        } else {

            data = ramadanData[currentSelectedIndex];

            const parts = data.date.split(' ');

            const monthStr = parts[0];

            const dayStr = parts[1].replace(',', '');

            const yearStr = parts[2];

            dateRef = new Date(`${monthStr} ${dayStr}, ${yearStr} 12:00:00`);

        }

        sehriTime = getTimeObj(data.sehri, dateRef);

        iftarTime = getTimeObj(data.iftar, dateRef);

        let nextPrayerTime = null;

        let prayerName = "";

        let targetDisplayTime = "";

        let statusText = "";

        let statusClass = "";

        document.getElementById('boxSehri').classList.remove('active');

        document.getElementById('boxIftar').classList.remove('active');

        if (!isViewingToday) {

            document.getElementById('countdownTimer').textContent = "00:00:00";

            document.getElementById('countdownLabel').textContent = "Selected Day";

            document.getElementById('statusBadge').textContent = data.dayName;

            document.getElementById('statusBadge').className = "status-badge";

            document.getElementById('boxIftar').classList.add('active');

            return;

        }

        if (now < sehriTime) {

            nextPrayerTime = sehriTime;

            prayerName = "Sehri Ends";

            targetDisplayTime = data.sehri;

            statusText = "Eating Time (Sehri)";

            statusClass = "";

            document.getElementById('boxSehri').classList.add('active');

        } else if (now >= sehriTime && now < iftarTime) {

            nextPrayerTime = iftarTime;

            prayerName = "Iftar Begins";

            targetDisplayTime = data.iftar;

            statusText = "Fasting in Progress";

            statusClass = "fasting";

            document.getElementById('boxIftar').classList.add('active');

        } else {

            document.getElementById('countdownTimer').textContent = "00:00:00";

            document.getElementById('countdownLabel').textContent = "Iftar Time Passed";

            document.getElementById('statusBadge').textContent = "Maghrib Time";

            document.getElementById('statusBadge').className = "status-badge iftar";

            document.getElementById('boxIftar').classList.add('active');

            return;

        }

        const diff = nextPrayerTime - now;

        document.getElementById('countdownTimer').textContent = formatTimeRemaining(diff);

        document.getElementById('countdownLabel').textContent = `Time for ${prayerName}`;

       

        const badge = document.getElementById('statusBadge');

        badge.textContent = statusText;

        badge.className = `status-badge ${statusClass}`;

    }

    function copyDua(type) {

        let text = "";

        let label = "";

        if(type === 'navait') {

            text = document.getElementById('duaSehriText').innerText;

            label = "Sehri";

        } else {

            text = document.getElementById('duaIftarText').innerText;

            label = "Iftar";

        }

        navigator.clipboard.writeText(text).then(() => {

            showToast(`Dua for ${label} copied!`);

        }).catch(err => {

            const textArea = document.createElement("textarea");

            textArea.value = text;

            document.body.appendChild(textArea);

            textArea.select();

            document.execCommand("copy");

            document.body.removeChild(textArea);

            showToast(`Dua for ${label} copied!`);

        });

    }

    function showToast(message) {

        const toast = document.getElementById('toast');

        const span = toast.querySelector('span');

        span.textContent = message;

       

        toast.classList.add('show');

       

        if (window.toastTimeout) clearTimeout(window.toastTimeout);

       

        window.toastTimeout = setTimeout(() => {

            toast.classList.remove('show');

        }, 3000);

    }

    document.addEventListener('DOMContentLoaded', () => {

        const todayStr = new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

       

        let foundToday = false;

        ramadanData.forEach((item, index) => {

            if (item.date === todayStr) {

                todayIndex = index;

                currentSelectedIndex = index;

                foundToday = true;

            }

        });

        if (!foundToday) {

            todayIndex = 0;

            currentSelectedIndex = 0;

        }

        renderList();

        updateCard();

        updateCountdown();

        setInterval(updateCountdown, 1000);

    });
